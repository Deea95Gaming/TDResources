stock Notif(playerid, message[], timer)
{
	new time=timer*1000;
	
	for(new i; i < 9; i++)
	{
		PlayerTextDrawShow(playerid, NotifTD[playerid][i]);
	}
	PlayerTextDrawSetString(playerid, NotifTD[playerid][1], message);
	SetTimerEx("HideNotif", time, false, "i", playerid);
	return 1;
}

forward HideNotif(playerid);
public HideNotif(playerid)
{
	for(new i; i < 9; i++)
	{
		PlayerTextDrawHide(playerid, NotifTD[playerid][i]);
	}
}

CMD:notifme(playerid, params[])
{
	new message[100], time;
	if(sscanf(params, "ds[100]", time, message))
		return SendClientMessage(playerid, -1, "[Usage] /notifme [duration on second] [message/text]");

	Notif(playerid, message, time);
	return 1;
}
